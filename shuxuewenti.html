# 快速幂

非递归版本
int p(int a,int b){
  int t,y;
  t=1; y=a;
  while (b!=0){
    if (b&1==1) t=t*y%MOD;
    y=y*y%MOD;
    b=b>>1;
  }
  return t;
}
